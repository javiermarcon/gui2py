2 """
3  sphinx.ext.autosummary
4  ~~~~~~~~~~~~~~~~~~~~~~
5 
6  Sphinx extension that adds an autosummary:: directive, which can be
7  used to generate function/method/attribute/etc. summary lists, similar
8  to those output eg. by Epydoc and other API doc generation tools.
9 
10  An :autolink: role is also provided.
11 
12  autosummary directive
13  ---------------------
14 
15  The autosummary directive has the form::
16 
17  .. autosummary::
18  :nosignatures:
19  :toctree: generated/
20 
21  module.function_1
22  module.function_2
23  ...
24 
25  and it generates an output table (containing signatures, optionally)
26 
27  ======================== =============================================
28  module.function_1(args) Summary line from the docstring of function_1
29  module.function_2(args) Summary line from the docstring
30  ...
31  ======================== =============================================
32 
33  If the :toctree: option is specified, files matching the function names
34  are inserted to the toctree with the given prefix:
35 
36  generated/module.function_1
37  generated/module.function_2
38  ...
39 
40  Note: The file names contain the module:: or currentmodule:: prefixes.
41 
42  .. seealso:: autosummary_generate.py
43 
44 
45  autolink role
46  -------------
47 
48  The autolink role functions as ``:obj:`` when the name referred can be
49  resolved to a Python object, and otherwise it becomes simple emphasis.
50  This can be used as the default role to make links 'smart'.
51 
52  :copyright: Copyright 2007-2011 by the Sphinx team, see AUTHORS.
53  :license: BSD, see LICENSE for details.
54 """

233  def get_items(self, names):
234  """Try to import the given names, and return a list of
235  ``[(name, signature, summary_string, real_name), ...]``.
236  """
237  env = self.state.document.settings.env
238 
239  prefixes = get_import_prefixes_from_env(env)
240 
241  items = []
242 
243  max_item_chars = 50
244 
245  for name in names:
246  display_name = name
247  if name.startswith('~'):
248  name = name[1:]
249  display_name = name.split('.')[-1]
250 
251  try:
252  real_name, obj, parent = import_by_name(name, prefixes=prefixes)
253  except ImportError:
254  self.warn('failed to import %s' % name)
255  items.append((name, '', '', name))
256  continue
257 
258  # NB. using real_name here is important, since Documenters
259  # handle module prefixes slightly differently
260  documenter = get_documenter(obj, parent)(self, real_name)
261  if not documenter.parse_name():
262  self.warn('failed to parse name %s' % real_name)
263  items.append((display_name, '', '', real_name))
264  continue
265  if not documenter.import_object():
266  self.warn('failed to import object %s' % real_name)
267  items.append((display_name, '', '', real_name))
268  continue
269 
270  # -- Grab the signature
271 
272  sig = documenter.format_signature()
273  if not sig:
274  sig = ''
275  else:
276  max_chars = max(10, max_item_chars - len(display_name))
277  sig = mangle_signature(sig, max_chars=max_chars)
278  sig = sig.replace('*', r'\*')
279 
280  # -- Grab the summary
281 
282  doc = list(documenter.process_doc(documenter.get_doc()))
283 
284  while doc and not doc[0].strip():
285  doc.pop(0)
286  m = re.search(r"^([A-Z][^A-Z]*?\.\s)", " ".join(doc).strip())
287  if m:
288  summary = m.group(1).strip()
289  elif doc:
290  summary = doc[0].strip()
291  else:
292  summary = ''
293 
294  items.append((display_name, sig, summary, real_name))
295

342 def mangle_signature(sig, max_chars=30):
343  """Reformat a function signature to a more compact form."""
344  s = re.sub(r"^\((.*)\)$", r"\1", sig).strip()
345 
346  # Strip strings (which can contain things that confuse the code below)
347  s = re.sub(r"\\\\", "", s)
348  s = re.sub(r"\\'", "", s)
349  s = re.sub(r"'[^']*'", "", s)
350 
351  # Parse the signature to arguments + options
352  args = []
353  opts = []
354 
355  opt_re = re.compile(r"^(.*, |)([a-zA-Z0-9_*]+)=")
356  while s:
357  m = opt_re.search(s)
358  if not m:
359  # The rest are arguments
360  args = s.split(', ')
361  break
362 
363  opts.insert(0, m.group(2))
364  s = m.group(1)[:-2]
365 
366  # Produce a more compact signature
367  sig = limited_join(", ", args, max_chars=max_chars-2)
368  if opts:
369  if not sig:
370  sig = "[%s]" % limited_join(", ", opts, max_chars=max_chars-4)
371  elif len(sig) < max_chars - 4 - 2 - 3:
372  sig += "[, %s]" % limited_join(", ", opts,
373  max_chars=max_chars-len(sig)-4-2)
374 
375  return u"(%s)" % sig

377 def limited_join(sep, items, max_chars=30, overflow_marker="..."):
378  """Join a number of strings to one, limiting the length to *max_chars*.
379 
380  If the string overflows this limit, replace the last fitting item by
381  *overflow_marker*.
382 
383  Returns: joined_string
384  """
385  full_str = sep.join(items)
386  if len(full_str) < max_chars:
387  return full_str
388 
389  n_chars = 0
390  n_items = 0
391  for j, item in enumerate(items):
392  n_chars += len(item) + len(sep)
393  if n_chars < max_chars - len(overflow_marker):
394  n_items += 1
395  else:
396  break
397 
398  return sep.join(list(items[:n_items]) + [overflow_marker])

439 def _import_by_name(name):
440  """Import a Python object given its full name."""
441  try:
442  name_parts = name.split('.')
443 
444  # try first interpret `name` as MODNAME.OBJ
445  modname = '.'.join(name_parts[:-1])
446  if modname:
447  try:
448  __import__(modname)
449  mod = sys.modules[modname]
450  return getattr(mod, name_parts[-1]), mod
451  except (ImportError, IndexError, AttributeError):
452  pass
453 
454  # ... then as MODNAME, MODNAME.OBJ1, MODNAME.OBJ1.OBJ2, ...
455  last_j = 0
456  modname = None
457  for j in reversed(range(1, len(name_parts)+1)):
458  last_j = j
459  modname = '.'.join(name_parts[:j])
460  try:
461  __import__(modname)
462  except:# ImportError:
463  continue
464  if modname in sys.modules:
465  break
466 
467  if last_j < len(name_parts):
468  parent = None
469  obj = sys.modules[modname]
470  for obj_name in name_parts[last_j:]:
471  parent = obj
472  obj = getattr(obj, obj_name)
473  return obj, parent
474  else:
475  return sys.modules[modname], None
476  except (ValueError, ImportError, AttributeError, KeyError), e:
477  raise ImportError(*e.args)
