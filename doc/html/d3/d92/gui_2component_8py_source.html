49 class ComponentBase(type):
50  "Component class constructor (creates metadata and register the component)"
51  def __new__(cls, name, bases, attrs):
52  super_new = super(ComponentBase, cls).__new__
53 
54  # Create the class.
55  new_class = super_new(cls, name, bases, attrs)
56 
57  specs = {}
58  # get specs of the base classes
59  for base in bases:
60  if hasattr(base, "_meta"):
61  specs.update(base._meta.specs)
62  # get all the specs
63  specs.update(dict([(attr_name, attr_value)
64  for attr_name, attr_value in attrs.items()
65  if isinstance(attr_value, Spec)]))
66  # insert a _meta attribute with the specs
67  new_class._meta = ComponentMeta(name, specs)
68 
69  # registry and return the new class:
70  if hasattr(new_class, "_registry"):
71  new_class._registry[name] = new_class
72  if name not in registry.ALL:
73  registry.ALL.append(name)
74  return new_class

299  "Raises/lower the component in the window hierarchy (Z-order/tab order)"
