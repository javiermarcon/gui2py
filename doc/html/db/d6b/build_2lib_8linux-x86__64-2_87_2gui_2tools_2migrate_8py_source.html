149 def migrate_window(bg):
150  "Take a pythoncard background resource and convert to a gui2py window"
151  ret = {}
152  for k, v in bg.items():
153  if k == 'type':
154  v = WIN_MAP[v]._meta.name
155  elif k == 'menubar':
156  menus = v['menus']
157  v = [migrate_control(menu) for menu in menus]
158  elif k == 'components':
159  v = [migrate_control(comp) for comp in v]
160  else:
161  k = SPEC_MAP['Widget'].get(k, k)
162  ret[k] = v
163  return ret
