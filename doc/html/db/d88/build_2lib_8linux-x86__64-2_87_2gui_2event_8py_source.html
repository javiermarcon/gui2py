24  cancel_default = False # flag to avoid default (generic) handlers

26  def __init__(self, name="", wx_event=None):
27  self.wx_event = wx_event
28  # retrieve wxPython event properties:
29  wx_obj = self.wx_event.GetEventObject()
30  # look for gui object (compound control with wx childs controls)
31  while wx_obj and not hasattr(wx_obj, "obj"):
32  wx_obj = wx_obj.Parent
33  self.target = wx_obj.obj if wx_obj else None
34  self.timestamp = wx_event.GetTimestamp()
35  # check if timestamp (wx report it only for mouse or keyboard)
36  if not self.timestamp:
37  self.timestamp = time.time() # create a new timestamp if not given
38  self.name = name # name (type), i.e.: "click"
39 
40  def prevent_default(self, cancel=True):
41  self.wx_event.Skip(not cancel)
42  self.cancel_default = cancel
