302 def find_autosummary_in_lines(lines, module=None, filename=None):
303  """Find out what items appear in autosummary:: directives in the
304  given lines.
305 
306  Returns a list of (name, toctree, template) where *name* is a name
307  of an object and *toctree* the :toctree: path of the corresponding
308  autosummary directive (relative to the root of the file name), and
309  *template* the value of the :template: option. *toctree* and
310  *template* ``None`` if the directive does not have the
311  corresponding options set.
312  """
313  autosummary_re = re.compile(r'^(\s*)\.\.\s+autosummary::\s*')
314  automodule_re = re.compile(
315  r'^\s*\.\.\s+automodule::\s*([A-Za-z0-9_.]+)\s*$')
316  module_re = re.compile(
317  r'^\s*\.\.\s+(current)?module::\s*([a-zA-Z0-9_.]+)\s*$')
318  autosummary_item_re = re.compile(r'^\s+(~?[_a-zA-Z][a-zA-Z0-9_.]*)\s*.*?')
319  toctree_arg_re = re.compile(r'^\s+:toctree:\s*(.*?)\s*$')
320  template_arg_re = re.compile(r'^\s+:template:\s*(.*?)\s*$')
321 
322  documented = []
323 
324  toctree = None
325  template = None
326  current_module = module
327  in_autosummary = False
328  base_indent = ""
329 
330  for line in lines:
331  if in_autosummary:
332  m = toctree_arg_re.match(line)
333  if m:
334  toctree = m.group(1)
335  if filename:
336  toctree = os.path.join(os.path.dirname(filename),
337  toctree)
338  continue
339 
340  m = template_arg_re.match(line)
341  if m:
342  template = m.group(1).strip()
343  continue
344 
345  if line.strip().startswith(':'):
346  continue # skip options
347 
348  m = autosummary_item_re.match(line)
349  if m:
350  name = m.group(1).strip()
351  if name.startswith('~'):
352  name = name[1:]
353  if current_module and \
354  not name.startswith(current_module + '.'):
355  name = "%s.%s" % (current_module, name)
356  documented.append((name, toctree, template))
357  continue
358 
359  if not line.strip() or line.startswith(base_indent + " "):
360  continue
361 
362  in_autosummary = False
363 
364  m = autosummary_re.match(line)
365  if m:
366  in_autosummary = True
367  base_indent = m.group(1)
368  toctree = None
369  template = None
370  continue
371 
372  m = automodule_re.search(line)
373  if m:
374  current_module = m.group(1).strip()
375  # recurse into the automodule docstring
376  documented.extend(find_autosummary_in_docstring(
377  current_module, filename=filename))
378  continue
379 
380  m = module_re.match(line)
381  if m:
382  current_module = m.group(2)
383  continue
384 
385  return documented
